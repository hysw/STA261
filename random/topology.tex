\section{Topology}

\subsection{Definitions}

\begin{description}
\item[metric]\phantomsection\label{def:metric}
	A function $d: X \times X \to \bR$ such that \hfill\\
	For all $x, y, z \in X$
	\begin{itemize}
	\item $d(x, y)\geq 0$
	\item $d(x, y) = 0 \iff x=y$
	\item $d(x, y) = d(y, x)$
	\item $d(x, z) \leq d(x, y) + d(y, z)$
	\end{itemize}

\item[$\epsilon$-neighborhood]$\lU(x;\epsilon)=\{y|d(x,y)<\epsilon\}$

\item[open set in metric space]\phantomsection\label{def:metric:openset}
	A set $U\subseteq X$ is said to be open in $X$ if\hfill\\
		$\forall x\in U\exists \epsilon>0 [\lU(x;\epsilon)\subseteq U]$\newline
	\extranote{note that finite intersections and arbitrary unions of open set are open set}

\item[closed set in metric space]\phantomsection\label{def:metric:closedset}
	A set contains all its limit point.\hfill\\
	\extranote{note that closed set is complement of open set in topology}
\end{description}
